/*
 * ====================================================================
 * CDS VIEW: ZPRAC_CDS_DEFAULT_PARAM
 * SQL VIEW: ZPRAC_SQL_DEFPAR
 * PURPOSE : CDS View parameters with DEFAULT values (optional params)
 * TOPIC   : Default parameter values — making parameters optional
 * ====================================================================
 *
 * KEY LEARNING POINTS:
 *   1. Parameters become OPTIONAL when they have a DEFAULT value
 *   2. If caller omits the parameter, the DEFAULT is used
 *   3. DEFAULT values can be:
 *      - Literal strings (must match target type)
 *      - Numeric literals
 *      - Date literals (YYYYMMDD)
 *      - $session.system_date (dynamic default!)
 *   4. All parameters with defaults are optional — mix of optional
 *      and mandatory parameters is allowed
 *   5. A view can have ONLY optional params (all with defaults)
 *   6. @Semantics annotations on fields enable OData/Analytics consumption
 *
 * USAGE IN ABAP:
 *   " With all parameters (override defaults)
 *   SELECT * FROM ZPRAC_CDS_DEFAULT_PARAM(
 *     p_currency = 'EUR',
 *     p_max_price = '5000' )
 *     INTO TABLE @DATA(lt_travel1).
 *
 *   " Without parameters (all defaults applied)
 *   SELECT * FROM ZPRAC_CDS_DEFAULT_PARAM( )
 *     INTO TABLE @DATA(lt_travel2).
 *   --> Equivalent to: currency_code = 'USD' AND max_price = 99999999
 * ====================================================================
 */
@AbapCatalog.sqlViewName: 'ZPRAC_SQL_DEFPAR'
@AbapCatalog.compiler.compareFilter: true
@AbapCatalog.preserveKey: true
@AccessControl.authorizationCheck: #CHECK
@EndUserText.label: 'Practice CDS - Parameters with default values (optional)'
define view ZPRAC_CDS_DEFAULT_PARAM
  with parameters
    p_currency  : abap.cuky DEFAULT 'USD',
    p_max_price : /dmo/total_price DEFAULT '99999999',
    p_min_date  : /dmo/begin_date DEFAULT '20200101'
  as select from /dmo/travel
{
  @Environment.systemField: #CLIENT
  key client            as Client,
  key travel_id         as TravelId,
      agency_id         as AgencyId,
      customer_id       as CustomerId,
      @Semantics.date: true
      begin_date        as BeginDate,
      @Semantics.date: true
      end_date          as EndDate,
      @Semantics.amount.currencyCode: 'CurrencyCode'
      booking_fee       as BookingFee,
      @Semantics.amount.currencyCode: 'CurrencyCode'
      total_price       as TotalPrice,
      currency_code     as CurrencyCode,
      description       as Description,
      status            as Status
}
where
      currency_code = $parameters.p_currency
  and total_price  <= $parameters.p_max_price
  and begin_date   >= $parameters.p_min_date
