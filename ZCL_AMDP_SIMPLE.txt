/*
 * ====================================================================
 * AMDP CLASS: ZCL_AMDP_SIMPLE
 * PURPOSE : AMDP with a simple input parameter — fetch travel by ID
 * TOPIC   : AMDP (ABAP Managed Database Procedure) — basic
 * ====================================================================
 *
 * WHAT IS AN AMDP?
 *   AMDP allows writing HANA database procedures directly in ABAP classes.
 *   Benefits over native CDS:
 *     ✓ Complex procedural logic (LOOPs, IF/ELSE, temp tables)
 *     ✓ Dynamic SQL via APPLY_FILTER
 *     ✓ Multiple result sets (when using PROCEDURE, not FUNCTION)
 *     ✓ HANA-specific features (PAL, SERIES, etc.)
 *
 * AMDP TYPES:
 *   BY DATABASE FUNCTION  → Single result set (used by Table Functions)
 *   BY DATABASE PROCEDURE → Multiple result sets (called from ABAP reports)
 *
 * KEY RULES:
 *   1. Class must implement IF_AMDP_MARKER_HDB
 *   2. Method must be static (CLASS-METHODS)
 *   3. Parameters passed by VALUE (IMPORTING VALUE(...))
 *   4. USING clause declares external CDS/tables accessible in SQLScript
 *   5. OPTIONS READ-ONLY for query-only procedures
 *
 * USAGE IN ABAP REPORT:
 *
 *   " Call via class name (static)
 *   zcl_amdp_simple=>get_details(
 *     EXPORTING
 *       im_travel_id      = '00000014'
 *     IMPORTING
 *       ex_travel_details = DATA(lt_travel_1) ).
 *
 *   " Call via object instance
 *   DATA(lo_amdp) = NEW zcl_amdp_simple( ).
 *   lo_amdp->get_details(
 *     EXPORTING
 *       im_travel_id      = '00000009'
 *     IMPORTING
 *       ex_travel_details = DATA(lt_travel_2) ).
 *
 * ====================================================================
 */
CLASS zcl_amdp_simple DEFINITION PUBLIC.

  PUBLIC SECTION.
    INTERFACES if_amdp_marker_hdb.

    " Local types for the return parameter
    TYPES: BEGIN OF ty_details,
             TravelId   TYPE /dmo/travel_id,
             AgencyId   TYPE /dmo/agency_id,
             CustomerId TYPE /dmo/customer_id,
             BookingFee TYPE /dmo/booking_fee,
           END OF ty_details,
           tt_details TYPE TABLE OF ty_details.

    " Method with single input parameter
    CLASS-METHODS: get_details
      IMPORTING
        VALUE(im_travel_id)      TYPE /dmo/travel_id
      EXPORTING
        VALUE(ex_travel_details) TYPE tt_details.

ENDCLASS.

CLASS zcl_amdp_simple IMPLEMENTATION.

  METHOD get_details
    BY DATABASE PROCEDURE FOR HDB LANGUAGE SQLSCRIPT
    OPTIONS READ-ONLY
    USING ZPRAC_CDS_SIMPLE.

    /*
     * Fetch data from CDS view filtered by input parameter.
     * In SQLScript, ABAP parameters are referenced with :PARAM_NAME (uppercase).
     */
    EX_TRAVEL_DETAILS =
      SELECT TravelId,
             AgencyId,
             CustomerId,
             BookingFee
      FROM  ZPRAC_CDS_SIMPLE
      WHERE TravelId = :IM_TRAVEL_ID;

  ENDMETHOD.

ENDCLASS.
