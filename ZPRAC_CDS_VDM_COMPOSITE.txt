 * ====================================================================

 * VDM LAYER POSITION:
 *   TABLE → BASIC (#BASIC) → COMPOSITE (#COMPOSITE) → CONSUMPTION (#CONSUMPTION)
 *
 * COMPOSITE VIEW CHARACTERISTICS:
 *   ✓ Built on #BASIC views (not directly on tables)
 *   ✓ Contains business logic: calculated fields, CASE, arithmetic
 *   ✓ May define new associations for downstream consumption
 *   ✓ @VDM.viewType: #COMPOSITE
 *   ✓ @Analytics.dataCategory: #CUBE or #FACT
 *   ✓ Named with C_ prefix in SAP (e.g. C_Travel_TP)
 *   ✓ Reused by multiple #CONSUMPTION views
 * ====================================================================
 */
@AbapCatalog.sqlViewName: 'ZPRAC_SQL_VDMCOM'
@AbapCatalog.compiler.compareFilter: true
@AbapCatalog.preserveKey: true
@AccessControl.authorizationCheck: #CHECK
@EndUserText.label: 'Practice CDS - VDM Composite View with business logic'
@VDM.viewType: #COMPOSITE
@Analytics.dataCategory: #CUBE
@ObjectModel.dataType: #COMPOSITE
define view ZPRAC_CDS_VDM_COMPOSITE
  as select from ZPRAC_CDS_VDM_BASIC as _Basic
{
  key _Basic.TravelId,
      _Basic.AgencyId,
      _Basic.CustomerId,
      @Semantics.date: true
      _Basic.BeginDate,
      @Semantics.date: true
      _Basic.EndDate,
      @Semantics.amount.currencyCode: 'CurrencyCode'
      _Basic.BookingFee,
      @Semantics.amount.currencyCode: 'CurrencyCode'
      _Basic.TotalPrice,
      _Basic.CurrencyCode,
      _Basic.Description,
      _Basic.Status,

  /* ── Business logic / Calculated Fields ── */
      @Semantics.amount.currencyCode: 'CurrencyCode'
      _Basic.TotalPrice - _Basic.BookingFee as NetTravelPrice,
      dats_days_between(_Basic.BeginDate, _Basic.EndDate) as TripDuration,
      concat(_Basic.Description, ' (', _Basic.Status, ')') as DescWithStatus,
      case
        when _Basic.TotalPrice > 5000 then 'High Value'
        when _Basic.TotalPrice > 1000 then 'Standard'
        else 'Budget'
      end                                              as ValueCategory
}
