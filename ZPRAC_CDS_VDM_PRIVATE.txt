 * SQL VIEW: ZPRAC_SQL_VDMPRI
 * PURPOSE : VDM Private View — internal helper / non-consumable
 * TOPIC   : @VDM.viewType: #PRIVATE — for internal use only
 * ====================================================================
 *
 * KEY LEARNING POINTS:
 *   1. #PRIVATE views are INTERNAL helper views — not consumed by
 *      end-user applications (Fiori, OData, Analytics)
 *   2. They are consumed by other CDS views (#BASIC, #COMPOSITE, etc.)
 *   3. Characteristics:
 *      ✓ Contains shared logic used by multiple public views
 *      ✓ Avoids code duplication across the CDS layer
 *      ✓ Is not exposed via OData (no @OData.publish)
 *      ✓ Named with P_ prefix in SAP standard
 *      ✓ Called "Private" because they are invisible to consumers
 *   4. Use #PRIVATE views for:
 *      - Common calculation patterns reused across views
 *      - Data transformation/complex joins used internally
 *      - SQLScript-based logic wrapped for reuse
 *      - Text retrieval logic for language-dependent fields
 *   5. #PRIVATE is the default viewType if no @VDM.viewType is specified
 *
 * VDM VIEW TYPE HIERARCHY:
 *   PRIVATE → BASIC → COMPOSITE → CONSUMPTION
 *   (Lowest)                      (Highest)
 * ====================================================================
 */
@AbapCatalog.sqlViewName: 'ZPRAC_SQL_VDMPRI'
@AbapCatalog.compiler.compareFilter: true
@AbapCatalog.preserveKey: true
@AccessControl.authorizationCheck: #NOT_REQUIRED
@EndUserText.label: 'Practice CDS - VDM Private View (internal helper)'
@VDM.viewType: #PRIVATE
@ObjectModel.dataType: #PRIVATE
define view ZPRAC_CDS_VDM_PRIVATE
  as select from /dmo/travel
{
  @Environment.systemField: #CLIENT
  key client            as Client,
  key travel_id         as TravelId,
      agency_id         as AgencyId,
      customer_id       as CustomerId,
      @Semantics.date: true
      begin_date        as BeginDate,
      @Semantics.date: true
      end_date          as EndDate,
      @Semantics.amount.currencyCode: 'CurrencyCode'
      booking_fee       as BookingFee,
      @Semantics.amount.currencyCode: 'CurrencyCode'
      total_price       as TotalPrice,
      currency_code     as CurrencyCode,
      description       as Description,
      status            as Status,

  /* ── Internal helper logic (shared across views) ── */
      @Semantics.amount.currencyCode: 'CurrencyCode'
      total_price - booking_fee as NetPrice,
      concat(description, ' - ', status) as CombinedInfo,
      dats_days_between(begin_date, end_date) as DurationDays,
      case
        when status = 'B' then 'Accepted'
        when status = 'N' then 'New'
        when status = 'C' then 'Completed'
        when status = 'X' then 'Cancelled'
        else 'Unknown'
      end as StatusText
}
